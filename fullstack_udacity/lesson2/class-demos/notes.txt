POSTGRES CONFIGS
-- login to postgres user
sudo -i -u postgres

-- check postgres version
pg_config --version

-- locate pg_config file
which pg_config

-- add postgres to path
export PATH=/usr/lib/postgresql/X.Y/bin/:$PATH

PSQL useful commands:
-- Conectando ao servidor remotamente via SSH
ssh -p 22 Administrator@192.168.100.57 
-- Conectando ao postgre sql como superadmin
sudo -u postgres psql
-- Listando todos os bancos de dados
psql -h 127.0.0.1 -p 5432 \l 

PSQL TERMINAL
-- conectando ao psql terminal
psql db_name --username=USERNAME

-- Listando todos os bancos de dados
\l 
-- Conectando ao banco de dados "name"
\c "database"
-- Listando todas as relações
\d
-- Listando todas as tabelas
\dt
-- Listando todas as colunas de determinada tabela
\d+ "table_name"
-- Realizando um select e salvando um arquivo na raiz do usuário linux
SELECT * FROM mediainfo \o essemeuteste2.txt	
-- Expandir os resultados (Para tabelas grandes) '
\x

-- Sair do terminal
\q
